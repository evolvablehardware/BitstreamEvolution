 74    def evaluate_fitness(self,population:tuple[Individual]) -> tuple[ tuple[Individual, Result[Fitness,Exception] ] ]:
 75       "This gets a population and outputs the fitness evaluation it performed using the hardware."
 76       results = []
 77       for individual in population:
 78           try:
 79               hardware_result = self.hardware.run_fitness_test(individual)
 80               fitness_result = Ok(Fitness()) if is_ok(hardware_result[1]) else hardware_result[1]
 81               results.append((individual, fitness_result))
 82           except Exception as e:
 83               results.append((individual, Err(e)))
 84       return tuple(results)
