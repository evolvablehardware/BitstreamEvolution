 33def v2verify_with(verify_function:Callable):
 34    "This is a decorator to attatch a verified function to it."
 35
 36    @ft.wraps(verify_function) # Maintain name information
 37    def verifier(func:Callable):
 38        verified = func
 39        verified.verify = verify_function
 40        # We could also potentially put this funtion into a list for testing.
 41        return verified
 42    
 43    return verifier

 47class V3Verifior:
 48    "When this class is passed to a function, it is requesting the function quickly verify it's arguments and functions it calls."
 49    def __init__(self):
 50        self.errors = []
 51
 52    def was_valid(self)->bool:
 53        return len(self.errors)
 54    
 55    def mark_invalid(self,message:str,function:Callable): #This could be an exception, too
 56        self.errors.append((message,function))
 57    
 58    def issues(self):
 59        if len(self.errors) == 0:
 60            print("Operations are Valid.")
 61
 62        for e in self.errors:
 63            print(f"Validation Error: '{e[0]}' in '{e[1].__name__}'")

 89def v2verify(higher_num:int, lower_num:int, mod_denom:int)->bool:
