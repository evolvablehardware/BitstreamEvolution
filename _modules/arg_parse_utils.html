13    group = arg_parser.add_mutually_exclusive_group(required=is_required)

15    ## create all flags enabling it
16    if len(flag_names['enable']) == 0:

22    ## create all flags disabling it
23    if len(flag_names['disable']) == 0:

25    else:
26        for flag in flag_names['disable']:
27            group.add_argument(flag,dest=feature_name, action='store_false')
